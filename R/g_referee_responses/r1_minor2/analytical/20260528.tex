\subsection{Ref's specification}

If $M(d) = \alpha + \beta |d|$, then the cumulative price effect is 
\begin{align*}
P(d) & = d \cdot M(d) = \alpha d + \beta |d| \cdot d \\
\Rightarrow P'(d) & = \alpha + 2 \beta |d| \numberthis \label{eq:marginal_P}
\end{align*}

\subsection{Inferring implied $P'(d)$ with standard errors}

\paragraph{Marginal impact}
We want to pin down marginal impact \eqref{eq:marginal_P} using our current specification. Using point estimates $M_1$ and $M_i$ at average bin demand levels $\bar{d}_1$ and $\bar{d}_i, i \in \{2,3\}$, 
the slope $\beta$ is estimated as:
\begin{equation}
\hat{\beta}_i = \frac{M_i - M_1}{\bar{d}_i - \bar{d}_1}
\end{equation}

Substituting $M_i = \alpha + \beta \bar{d}_i$ and $\hat{\beta}_i$ into the marginal impact equation at $\bar{d}_i$:
\begin{equation}
P'(\bar{d}_i) = M_i + \bar{d}_i \left( \frac{M_i - M_1}{\bar{d}_i - \bar{d}_1} \right)
\end{equation}

Defining the scaling constant $k_i = \frac{\bar{d}_i}{\bar{d}_i - \bar{d}_1}$, we express the 
marginal impact as a linear combination of the regression coefficients:
\begin{equation}
P'(\bar{d}_i) = (1 + k_i)M_i - k_i M_1
\end{equation}

\paragraph{Standard error via Delta method}
To compute the standard error of the marginal impact, let $\hat{\theta} = [M_1, M_i]^T$ 
with covariance matrix $\Sigma$:
\begin{equation}
\Sigma = \begin{bmatrix} \text{Var}(M_1) & \text{Cov}(M_1, M_i) \\ \text{Cov}(M_1, M_i) & \text{Var}(M_i) \end{bmatrix}
\end{equation}

The marginal impact $P'(\bar{d}_i)$ is a linear function $g(\hat{\theta})$. 
The gradient vector $\nabla g$ is:
\begin{equation}
\nabla g = \begin{bmatrix} \frac{\partial P'}{\partial M_1} \\ \frac{\partial P'}{\partial M_i} \end{bmatrix} = \begin{bmatrix} -k_i \\ 1 + k_i \end{bmatrix}
\end{equation}

Applying the Delta Method, the variance of the marginal impact is:
\begin{equation}
\text{Var}(P'(\bar{d}_i)) = \nabla g^T \Sigma \nabla g
\end{equation}
